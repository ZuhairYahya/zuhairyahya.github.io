Automation Deploy Kubernetes Cluster Using Ansible

Mon, Jan 1, 0001 One-minute read

Pada artikel ini saya akan menjelaskan mengenai cara automation deploy kubernetes menggunakan ansible

Install Ansible

  1. Update packges
sudo apt-get update
  1. Install properties
sudo apt-get install software-properties-common -y
  1. Install ansible
sudo apt-get install ansible -y

Deploy kubernetes

  1. Create inventory
[master]
master ansible_host=10.0.0.4 ansible_user=master

[worker]
worker ansible_host=10.0.0.5 ansible_user=worker

[all:vars]
ansible_ssh_private_key_file=~/.ssh/id_rsa
  1. Create ansible.cfg
vi ansible.cfg

[defaults]
inventory = ./inventory
  1. Create ansible playbook for kubernetes dependencies
vi kube-dependencies.yml

- hosts: all
  become: yes
  tasks:
    - name: Install Docker
      apt:
        name: docker.io
        state: present
        update_cache: true
    
    - name: start Docker
      service:
        name: docker
        state: started
        enabled: true
    
    - name: Install APT Transport HTTPS
      apt:
        name: apt-transport-https
        state: present

    - name: add kubernetes apt-key
      apt_key:
        url: https://packages.cloud.google.com/apt/doc/apt-key.gpg
        state: present

    - name: add kubernetes apt repository
      apt_repository:
        repo: deb http://apt.kubernetes.io/ kubernetes-xenial main
        state: present
        filename: 'kubernetes.list'
        
    - name: install kubectl
      apt:
        name: kubectl=1.19.0-00
        state: present
        force: yes
    
    - name: install kubelet
      apt:
        name: kubelet=1.19.0-00
        state: present
        force: yes
        update_cache: true
    
    - name: install kubeadm
      apt:
        name: kubeadm=1.19.0-00
        state: present
        force: yes
    
    - name: Start kubelet
      service:
        name: kubelet
        state: restarted
        enabled: true
  1. Jalankan ansible playbook
ansible-playbook kube-dependencies.yml
  1. Create ansible playbook for master
vi master.yml

- hosts: master
  become: yes
  tasks:
    - name: Init docker user
      shell: usermod -aG docker $USER

    - name: Remove swap
      shell: swapoff -a
    
    - name: Initiate cluster
      shell: kubeadm init --pod-network-cidr=10.244.0.0/16 >> initialized.txt
      args:
        chdir: $HOME
        creates: initialized.txt
    
    - name: Enabled Docker
      service:
        name: docker
        enabled: true
    
    - name: Enabled Docker
      service:
        name: docker
        state: restarted

    - name: Create .kube directory
      become: yes
      become_user: master
      file:
        path: $HOME/.kube
        state: directory
        mode: 0755
    
    - name: copy admin.conf to kube config
      copy:
        src: /etc/kubernetes/admin.conf
        dest: /home/master/.kube/config
        remote_src: yes
        owner: master
    
    - name: Install pod network
      become: yes
      become_user: master
      shell: kubectl apply -f https://raw.githubusercontent.com/coreos/flannel/master/Documentation/kube-flannel.yml >> pod_network_setup.txt
      args:
        chdir: $HOME
        creates: pod_network_setup.txt
  1. Jalankan ansible playbook for master.yml
ansible-playbook master.yml
  1. Create ansible playbook for worker.yml
vi worker.yml
  1. Jalankan ansible playbook for worker.yml
- hosts: master
  become: yes
  gather_facts: false
  tasks:
    - name: get join command
      shell: kubeadm token create --print-join-command
      register: join_command_raw
    - name: set join command
      set_fact:
        join_command: "{{ join_command_raw.stdout_lines[0] }}"
 
- hosts: worker
  become: yes
  tasks:
    - name: remove swap
      shell: "swapoff -a"
    - name: join cluster	
      shell: "{{ hostvars['master'].join_command }} >> node_joined.txt"
      args:
        chdir: $HOME
        creates: node_joined.txt
